####		CONFIGURATION FILE FORMAT GUIDE			####



	DIRECTIVE			|		VALUE			|		ENFORCEMENT
						|						|
server
{
	listen 					<PORT>					One-and-done
	root 					<URL>					One-and-done
	error_page 				<CODE> <URL> 			Multiple entries accepted
	autoindex				<on|off>				Last-seen-wins
	upload_path				<URL>					One-and-done
	allowed_methods			<GET|POST|DELETE>		One-and-done
	client_max_body_size	<SIZE&UNIT>				One-and-done	Note: Units are 'k', 'm', or 'g'. Mays accepted.
	enable_reuse_addr		<on|off>				Last_seen-wins
									
	location 				<PATH>
	{
		root 				<URL>					One-and-done	
		allowed_methods		<GET|POST|DELETE>		One-and-done
		autoindex			<on|off>				Last-seen-wins
		return 				<CODE> <URL>			One-and-done.	Note: Two LocationConfig elements in one directive.
		index 				<FILE1> <FILE2> ...		One-and-done
		upload_path 		<URL>					One-and-done	
	}
}
